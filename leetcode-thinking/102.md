```js
/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number[][]}
 */
var levelOrder = function(root) {
  if(!root)return []
  let res=[]
  let aux=[root]
  while(aux.length>0){
    let len=aux.length
    let vals=[]
    for(let i=0;i<len;i++){
      let node=aux.shift()
      vals.push(node.val)
      if(node.left)aux.push(node.left)
      if(node.right)aux.push(node.right)
    }
    res.push(vals)
  }
  return res
};
```
